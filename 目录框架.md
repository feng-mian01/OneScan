security_scan_tool/          # 项目根目录
├── bin/                     # 可执行文件/第三方工具目录（存放Nmap/Dirsearch等）
│   ├── nmap/                # Nmap Windows版（建议放绿色版，避免用户装环境）
│   ├── dirsearch/           # Dirsearch源码（Python版，直接调用）
│   └── tools_readme.md      # 工具说明（比如版本、使用前提）
├── config/                  # 配置文件目录
│   ├── fofa_config.ini      # FOFA API配置（token、邮箱，可让用户自行填写）
│   ├── scan_config.ini      # 扫描默认配置（比如默认端口、字典路径）
│   └── log_config.ini       # 日志配置（日志级别、保存路径）
├── core/                    # 核心功能模块（业务逻辑）
│   ├── __init__.py          # 让core成为Python包
│   ├── asset_collector.py   # 资产收集模块（FOFA/爱企查对接）
│   ├── port_scanner.py      # 端口扫描模块（调用Nmap）
│   ├── dir_scanner.py       # 目录扫描模块（调用Dirsearch）
│   └── result_parser.py     # 结果解析模块（统一解析各工具输出）
├── data/                    # 数据存储目录
│   ├── dicts/               # 扫描字典（放常用的dirsearch字典）
│   │   ├── common.txt       # 通用目录字典
│   │   └── sensitive.txt    # 敏感路径字典
│   ├── logs/                # 日志文件（自动生成）
│   │   └── app.log          # 程序运行日志
│   └── outputs/             # 导出结果目录（自动生成）
│       ├── excel/           # Excel导出结果
│       └── txt/             # TXT导出结果
├── ui/                      # 界面模块（PyQt5相关）
│   ├── __init__.py
│   ├── main_window.py       # 主窗口（核心界面）
│   ├── asset_ui.py          # 资产收集子界面
│   ├── port_ui.py           # 端口扫描子界面
│   ├── dir_ui.py            # 目录扫描子界面
│   └── resource/            # 界面资源（可选）
│       ├── icon.ico         # 程序图标
│       └── style.qss        # 界面样式（美化用）
├── utils/                   # 通用工具函数（复用代码）
│   ├── __init__.py
│   ├── api_utils.py         # API调用工具（FOFA/爱企查请求）
│   ├── cmd_utils.py         # 命令行调用工具（调用Nmap/Dirsearch）
│   ├── file_utils.py        # 文件操作工具（导出Excel/TXT）
│   └── log_utils.py         # 日志工具（记录程序运行日志）
├── main.py                  # 程序入口文件（启动主窗口）
├── requirements.txt         # 依赖包清单（PyQt5、requests、pandas等）
├── README.md                # 项目说明（使用方法、环境要求、功能介绍）
└── build.py                 # 打包脚本（用PyInstaller打包成exe）
